def sort_insertion(arr,start,end):
    for j in range(start+1,end+1):
        key = arr[j]
        i = j - 1
        while (i >= 0) and (arr[i] > key):
            arr[i+1] = arr[i]
            i -= 1
        arr[i+1] = key


def merge(arr,p,q,r):
    leftArr = arr[p:q+1]
    rightArr = arr[q+1:r+1]

    leftArr.append(float('inf'))
    rightArr.append(float('inf'))

    i = 0
    j = 0
    for k in range(p,r+1):
        if leftArr[i] <= rightArr[j]:
            arr[k] = leftArr[i]
            i += 1
        else:
            arr[k] = rightArr[j]
            j += 1


def sort_merge(arr,p,r):
    if p < r:
        q = (p+r)//2
        sort_merge(arr,p,q)
        sort_merge(arr,q+1,r)
        merge(arr,p,q,r)
    else:
        return

    i = 0
    while i < n:
        list.append(random.randint(0,MAX_INT_NUMBER))
        i += 1
    return list

import time

# @param fun    
# @param n       1e4->10000
# @param k      

def returnCalculateRes_avg(fun, n, k):
    sum = 0
    i = 0
    while i < k:
        list = generateTestSe_Int(n)
        t1 = time.perf_counter_ns()
        fun(list,0,len(list)-1)
        t2 = time.perf_counter_ns()
        sum += (t2 - t1)/1e6
        i += 1
    return sum/k

import math

def calculateTheoryTime_insertion(t,kArr):
    arr = []
    for i in kArr:
        res = math.pow(i/kArr[0],2) * t
        arr.append(res)
    return arr

def calculateTheoryTime_merge(t,kArr):
    arr = []
    for i in kArr:
        res = i/kArr[0] * math.log(i,kArr[0]) * t
        arr.append(res)
    return arr
