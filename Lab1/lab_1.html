# @param    arr     
# @param    start   index0
# @param    end     index
def sort_insertion(arr,start,end): # 
    for j in range(start+1,end+1):
        key = arr[j]
        i = j - 1
        while (i >= 0) and (arr[i] > key):
            arr[i+1] = arr[i]
            i -= 1
        arr[i+1] = key


# @param    arr     
# @param    p       index0
# @param    q       index
# @param    r       index
def merge(arr,p,q,r):   # 
    leftArr = arr[p:q+1]
    rightArr = arr[q+1:r+1]

    leftArr.append(float('inf'))
    rightArr.append(float('inf'))

    i = 0
    j = 0
    for k in range(p,r+1):
        if leftArr[i] <= rightArr[j]:
            arr[k] = leftArr[i]
            i += 1
        else:
            arr[k] = rightArr[j]
            j += 1


# @param    arr     
# @param    p       index0
# @param    r       index
def sort_merge(arr,p,r):    # 
    if p < r:
        q = (p+r)//2
        sort_merge(arr,p,q)
        sort_merge(arr,q+1,r)
        merge(arr,p,q,r)
    else:
        return

    return list

import time

# @param fun    
# @param n       1e4->10000
# @param k      

def returnCalculateRes_avg(fun, n, k):
    sum = 0
    i = 0
    while i < k:
        list = generateTestSe_Int(n)
        t1 = time.perf_counter_ns()
        fun(list,0,len(list)-1)
        t2 = time.perf_counter_ns()
        sum += (t2 - t1)/1e6
        i += 1
    return sum/k

理论时间转换公式推导¶

插入排序¶

$$ 1 \cdot 10^{4}t_{1}n_{1} = 10^{4} t_{1} = n_{1}^{2} \cdot t \newlinen_{2} t_{2}=n_{2}^{2} \cdot t = (\frac{n_{2}}{n_{1}}) ^ {2} \cdot t_{1} \newline t_{1} n_{2} = {k \cdot 10^{4},k \in (1,2,3,4,5)} $$

归并排序¶

$$ 1 \cdot 10^{4}t_{1}n_{1} = 10^{4} t_{1} = n_{1} \cdot \log_2{n_{1}} \cdot t \newlinen_{2} t_{2}=n_{2} \cdot \log_2{n_{2}} \cdot t = (\frac{n_{2}}{n_{1}}) \times (\frac{\log_2{n_{2}}}{\log_2{n_{1}}}) \cdot t_{1} = (\frac{n_{2}}{n_{1}}) \times (\log_{n_{1}}{n_{2}}) \cdot t_{1} \newline t_{1} n_{2} = {k \cdot 10^{4},k \in (1,2,3,4,5)} $$

import math

def calculateTheoryTime_insertion(t,kArr):
    arr = []
    for i in kArr:
        res = math.pow(i/kArr[0],2) * t
        arr.append(res)
    return arr

def calculateTheoryTime_merge(t,kArr):
    arr = []
    for i in kArr:
        res = i/kArr[0] * math.log(i,kArr[0]) * t
        arr.append(res)
    return arr
